  {\large\bfseries 外包采购价格的决定因素：回归结果汇总}\\[4pt]
  {\normalsize 2026-05-28}
\end{center}

\vspace{4pt}
\noindent\textbf{样本说明：} 2017年增值税发票数据，限定为\textbf{外包产品}——同一企业在同一年内对同一产品既有采购记录（\texttt{firm\_buy}）又有销售记录（\texttt{firm\_sell}）。\textbf{中介企业（外包比例 $>90\%$ 的纯转卖商）已全部剔除}，所有表格使用同一样本。因变量为外包采购单价对数 $\ln p_{fjct}$，对应结构模型 Step 2 的外包成本方程（样本限定 $Q^B_{fjt}>0$）。标准误聚类到企业层面，括号内为标准误。$^{*}p<0.10$，$^{**}p<0.05$，$^{***}p<0.01$。

\section*{表一：基准回归（逐步加入固定效应）}

\noindent\textbf{回归方程：}

\vspace{-4pt}
\begin{equation}
  \ln p_{fjct}
    = \underbrace{\gamma_1\ln\mathrm{Output}_{ft}
                + \gamma_2\ln K_{ft}
                + \gamma_3 n\mathrm{Prod}_{ft}}_{\text{企业特征 }x_{ft}}
    + \underbrace{\lambda_1\ln n^B_{jct}
                + \lambda_2\ln n^S_{jct}
                + \lambda_3\ln q^m_{jct}
                + \lambda_4\ln\bar{p}^m_{jct}}_{\text{市场条件 }z_{jct}}
    + \delta_f^{[k]} + \delta_j^{[k]} + \delta_c^{[k]}
    + \varepsilon_{fjct}
  \label{eq:baseline}
\end{equation}

企业特征变量（$x_{ft}$）在加入企业固定效应 $\delta_f$ 后被完全吸收，仅在 OLS 规格（列1--2）中可识别。市场条件变量（$z_{jct}$）在所有规格中均可识别。固定效应跨列逐步加入。

\section*{表二：需求侧 vs 供给侧分解}

\noindent\textbf{回归方程：}

\vspace{-4pt}
\begin{equation}
  \ln p_{fjct}
    = \underbrace{\lambda_1\ln n^B_{jct}
                + \lambda_3\ln q^m_{jct}}_{\text{需求侧（Demand）}}
    + \underbrace{\lambda_2\ln n^S_{jct}}_{\text{供给侧（Supply）}}
    + \underbrace{\lambda_4\ln\bar{p}^m_{jct}}_{\text{市场均价}}
    + \delta_f + \delta_j + \delta_c
    + \varepsilon_{fjct}
  \label{eq:demand_supply}
\end{equation}

各列逐步加入需求侧、供给侧和市场均价变量，全程保留 Firm + Product + City FE。需求侧：买方数 $n^B$ 与市场总量 $q^m$；供给侧：卖方数 $n^S$；市场均价 $\bar{p}^m$ 作为价格水平锚点。

\section*{表三：产品相似度（结构模型 $S_{mj}$, $C_{mj}$）}

\noindent\textbf{回归方程：}

\vspace{-4pt}
\begin{equation}
  \ln p_{fjct}
    = \boldsymbol{\lambda}'\mathbf{z}_{jct}
    + \beta_1 S_{mj}
    + \beta_2 C_{mj}
    + \delta_f + \delta_j + \delta_c
    + \varepsilon_{fjct}
  \label{eq:similarity}
\end{equation}

其中 $\mathbf{z}_{jct} = (\ln n^B_{jct},\ \ln n^S_{jct},\ \ln q^m_{jct},\ \ln\bar{p}^m_{jct})'$；$S_{mj}$（\texttt{input\_similarity}）为企业核心产品 $m$ 与外包产品 $j$ 的投入结构相似度；$C_{mj}$（\texttt{output\_similarity}）为产出结构相似度。两者来自 \texttt{full\_data.dta}，按各企业主产品（生产产值最大）预先计算。列(1)为基准（无相似度），列(2)-(4)逐步加入。

\section*{表四：企业规模 $\times$ 市场条件交互}

\noindent\textbf{回归方程：}

\begin{equation}
  \ln p_{fjct}
    = \boldsymbol{\lambda}'\mathbf{z}_{jct}
    + \mu_1\bigl(\ln\mathrm{Output}_{ft}\cdot\ln n^B_{jct}\bigr)
    + \mu_2\bigl(\ln\mathrm{Output}_{ft}\cdot\ln n^S_{jct}\bigr)
    + \mu_3\bigl(\ln\mathrm{Output}_{ft}\cdot\ln\bar{p}^m_{jct}\bigr)
